Flask-OtpAuth

>>> from flask_otpauth import OtpAuth
>>> SECRET = 'secret'
>>> auth = OtpAuth(app)

>>> auth.hotp(SECRET)
{{ hotp(SECRET) }}
>>> auth.valid_hotp(SECRET,
) {% for cat, msg in get_flashed_messages(with_categories=True) if cat == 'hotp' %}
